Это API для регистрации (изменения, удаления) пользователей.

GET / - документация;

OPTIONS / - допустимые запросы;

GET /usr/ - список всех пользователей

аргументы - count (число пользователей, которое будет выведено, по умолчанию все пользователи), offset (отступ, начиная с которого будет выведено count пользователей, по умолчанию 0);

POST /usr/ - добавление пользователя

аргументы - name,surname (обязательные);

DELETE /usr/ - удаление всех пользователей с этими данными

аргументы - name,surname (обязательные)

PUT /usr/ - изменение фамилии у пользователя на другую (аналогично с именем)

аргументы - name, surname(обязательные), newname или newsurname
